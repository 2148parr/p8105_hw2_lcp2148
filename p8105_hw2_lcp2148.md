``` r
baby_names = read_csv(file = "./hw2_data/Popular_Baby_Names.csv", col_types = "dcccdd")  %>%
  janitor::clean_names() %>%
  distinct() %>%
  mutate(childs_first_name = tolower(childs_first_name)) %>%
  mutate(gender = tolower(gender)) %>%
```
